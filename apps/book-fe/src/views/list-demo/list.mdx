import { Avatar, Divider, List, Tabs } from '@youknown/react-ui/src'

### 列表大小

export function ListSize() {
  const [size, setSize] = useState('small')
  return (
    <>
      <Tabs
        className="mb-8px"
        type="segment"
        value={size}
        onChange={setSize}
        tabList={[
          { key: 'small', name: 'Small' },
          { key: 'medium', name: 'Medium' },
          { key: 'large', name: 'Large' }
        ]}
      />
      <List size={size}>
        <List.Item>List item 1</List.Item>
        <List.Item>List item 2</List.Item>
        <List.Item>List item 2</List.Item>
      </List>
    </>
  )
}

<ListSize />

## API

### List Props

| 属性     | 说明       | 类型                             | 默认值     |
| -------- | ---------- | -------------------------------- | ---------- |
| bordered | 是否有边框 | `boolean`                        | `false`    |
| size     | 大小       | `'small' \| 'medium' \| 'large'` | `'medium'` |

### List Item Props

| 属性     | 说明       | 类型                             | 默认值     |
| -------- | ---------- | -------------------------------- | ---------- |
| bordered | 是否有边框 | `boolean`                        | `false`    |
| size     | 大小       | `'small' \| 'medium' \| 'large'` | `'medium'` |
| prefix   | 前缀图标   | `React.ReactNode`                | -          |
| suffix   | 后缀图标   | `React.ReactNode`                | -          |
