### 外部控制

export function InputRef() {
  const inputRef = useRef(null)
  return (
    <Space>
      <Input placeholder="Forward ref" ref={inputRef} />
      <Button
        primary
        onClick={() => {
          inputRef.current?.focus()
        }}
      >
        Manual focus
      </Button>
      <Button
        primary
        onClick={() => {
          inputRef.current?.blur()
        }}
      >
        Manual blur
      </Button>
    </Space>
  )
}

<InputRef />

## API

### Input Props

| 属性         | 说明             | 类型                             | 默认值     |
| ------------ | ---------------- | -------------------------------- | ---------- |
| type         | 输入框类型       | `string`                         | `'text'`   |
| size         | 大小             | `'small' \| 'medium' \| 'large'` | `'medium'` |
| round        | 是否圆角         | `boolean`                        | `false`    |
| bordered     | 有边框           | `boolean`                        | `true`     |
| outline      | 有外框           | `boolean`                        | `true`     |
| disabled     | 是否禁用         | `boolean`                        | `false`    |
| prefix       | 头部             | `React.ReactNode`                | -          |
| suffix       | 底部             | `React.ReactNode`                | -          |
| allowClear   | 是否展示清空按钮 | `boolean`                        | `false`    |
| autoFocus    | 是否自动聚焦     | `boolean`                        | `false`    |
| defaultValue | 默认值           | `string`                         | `''`       |
| value        | 值               | `string`                         | -          |
| onChange     | 值变化回调函数   | `(value: string) => void`        | -          |
| onEnter      | 是否自动聚焦     | `(value: string) => void`        | -          |

### Input Textarea Props

| 属性         | 说明           | 类型                      | 默认值     |
| ------------ | -------------- | ------------------------- | ---------- |
| minRows      | 输入框类型     | `number`                  | `1`        |
| maxRows      | 输入框类型     | `number`                  | `Infinity` |
| autosize     | 是否圆角       | `boolean`                 | `false`    |
| bordered     | 有边框         | `boolean`                 | `true`     |
| outline      | 有外框         | `boolean`                 | `true`     |
| disabled     | 是否禁用       | `boolean`                 | `false`    |
| autoFocus    | 是否自动聚焦   | `boolean`                 | `false`    |
| defaultValue | 默认值         | `string`                  | `''`       |
| value        | 值             | `string`                  | -          |
| onChange     | 值变化回调函数 | `(value: string) => void` | -          |
| onEnter      | 是否自动聚焦   | `(value: string) => void` | -          |
