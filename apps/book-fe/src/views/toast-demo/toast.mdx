### 基本用法

export function BasicUsage() {
  return (
    <Space>
      <Button
        onClick={() => {
          Toast({ content: 'Hi' })
        }}
      >
        Show basic toast
      </Button>
      <Button
        onClick={() => {
          Toast({
            content: (
              <span>
                <strong>Penguins</strong> are <span className="color-red">birds</span>.
              </span>
            )
          })
        }}
      >
        Toast content element
      </Button>
    </Space>
  )
}

### 进阶用法

export function Advance() {
  return (
    <Space>
      <Button
        onClick={() => {
          const inst = Toast({
            content: 'Click me',
            duration: 10000,
            className: 'cursor-pointer',
            onClick() {
              console.log('clicked')
              inst.close()
            },
            onClose() {
              console.log('closed')
            }
          })
        }}
      >
        Close toast
      </Button>
      <Button
        onClick={() => {
          const inst = Toast({
            content: 'Click me',
            duration: 5000,
            className: 'cursor-pointer',
            onClick() {
              console.log('clicked')
              inst.update({
                content: 'Hello',
                className: ''
              })
            },
            onClose() {
              console.log('closed')
            }
          })
        }}
      >
        Update toast
      </Button>
    </Space>
  )
}

<Advance />

## API

### Toast Options

| 参数      | 说明             | 类型                          | 默认值 |
| --------- | ---------------- | ----------------------------- | ------ |
| content   | 内容             | `React.ReactNode`             | -      |
| duration  | 持续时间（毫秒） | `number`                      | `2000` |
| icon      | 图标             | `React.ReactNode`             | -      |
| className | 样式class        | `string`                      | -      |
| style     | 样式             | `React.CSSProperties`         | -      |
| onClick   | 点击事件         | `(event: MouseEvent) => void` | -      |
| onClose   | 关闭回调函数     | `() => void`                  | -      |
