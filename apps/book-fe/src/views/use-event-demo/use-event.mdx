# useEvent

## 代码演示

import { useEvent } from '@youknown/react-hook/src'
import { Toast, Button } from '@youknown/react-ui/src'
import { useState, useCallback } from 'react'

确保函数地址永远不会变化

---

### 基本用法

export function BasicUsage() {
  const [count, setCount] = useState(0)

  const callbackFn = useCallback(() => {
    Toast.info(`Current count is ${count}`)
  }, [count])

  const eventFn = useEvent(() => {
    Toast.info(`Current count is ${count}`)
  })
  return (
    <>
      <p>count: {count}</p>
      <Button
        onClick={() => {
          setCount((c) => c + 1)
        }}
      >
        Add Count
      </Button>
      <div className='mt-16px'>
        <Button onClick={callbackFn}>
          call callbackFn
        </Button>
        <Button onClick={eventFn} className='ml-8px'>
          call event
        </Button>
      </div>
    </>
  )
}

## API

### Type

`(fn: (...arg: any) => void) => (...arg: any) => any`

### Params

| 参数       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| fn    | 需要持久化的函数           | `(...arg: any) => any`         | -  |

### Result

| 参数       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| fn    | 引用地址永远不会变化的函数           | `(...arg: any) => any`         | -  |
