# useUpdate

## 代码演示

import { useUpdate } from '@youknown/react-hook/src'
import { Button } from '@youknown/react-ui/src'

强制重新渲染组件。

---

### 基本用法

export function BasicUsage() {
  const update = useUpdate()
  return (
    <>
      <div>Time: {Date.now()}</div>
      <Button onClick={update} style={{ marginTop: 8 }}>
        update
      </Button>
    </>
  )
}
