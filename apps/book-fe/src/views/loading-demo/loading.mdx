### 容器类型的加载中

export function LoadingContainer() {
  const [spinning, setSpinning] = useState(true)
  return (
    <Space direction="vertical">
      <Button
        onClick={() => {
          setSpinning(p => !p)
        }}
      >
        {spinning ? 'Cancel' : 'Start'} spinning
      </Button>
      <Space>
        <Loading spinning={spinning} />
        <Loading spinning={spinning}>
          <Card header="Loading wrapper">Content</Card>
        </Loading>
        <Loading bordered spinning={spinning}>
          <Card header="Bordered spinning">Content</Card>
        </Loading>
        <Loading icon={<TbLoader />} spinning={spinning}>
          <Card header="Custom spinning icon">Content</Card>
        </Loading>
      </Space>
    </Space>
  )
}

<LoadingContainer />

## API

### Loading Props

| 属性        | 说明   | 类型                             | 默认值   |
| ----------- | ------ | -------------------------------- | -------- |
| spinning    | 加载中 | `boolean`                        | `false`  |
| bordered    | 有边框 | `boolean`                        | `false`  |
| size        | 大小   | `'small' \| 'medium' \| 'large'` | `medium` |
| icon        | 图标   | `React.ReactNode`                | -        |
| description | 描述   | `React.ReactNode`                | -        |
