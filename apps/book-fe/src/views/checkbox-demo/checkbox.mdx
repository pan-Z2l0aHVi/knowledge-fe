### 复选框组

export function CheckboxGroup() {
  const [groupVal, setGroupVal] = useState([2])
  const [dir, setDir] = useState('horizontal')
  return (
    <>
      <Tabs
        type="segment"
        tabList={[
          { key: 'horizontal', name: 'Horizontal' },
          { key: 'vertical', name: 'Vertical' }
        ]}
        value={dir}
        onChange={setDir}
      />
      <Divider />
      <Checkbox.Group
        direction={dir}
        options={[
          { label: 0, child: 'Cherry' },
          { label: 1, child: 'Apple' },
          { label: 2, child: 'Pear' },
          { label: 3, child: 'Orange', disabled: true }
        ]}
        value={groupVal}
        onChange={setGroupVal}
      />
    </>
  )
}

## API

### Checkbox Props

| 属性       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| disabled   | 禁用状态           | `boolean`         | `false`  |
| size       | 大小               | `'small'`         | `'small' \| 'medium' \| 'large'`  |
| label    | 标识id           | `string \| number`         | -  |
| defaultValue    | 默认值           | `boolean`         | `false`  |
| value    |  是否选中          | `boolean`         | -  |
| onChange    |  选中事件回调          | `(value: boolean) => void`         | -  |

### Checkbox.Group Props

| 属性       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| disabled   | 禁用状态           | `boolean`         | `false`  |
| size       | 大小               | `'small'`         | `'small' \| 'medium' \| 'large'`  |
| defaultValue    | 默认值           | `(string \| number)[]`         | `[]`  |
| value    | 选中值           | `(string \| number)[]`         | -  |
| onChange    | 选中事件回调           | `(value: (string \| number)[]) => void`         | -  |
| direction    | 方向           | `'vertical' \| 'horizontal'`         | `'horizontal'`  |
| options    | 可选项列表           | `Options`         | `[]`  |

### Options

| 参数       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| disabled   | 禁用状态           | `boolean`         | `false`  |
| label    | 标识id           | `string \| number`         | -  |
| child    | 子元素           | `React.ReactNode`         | -  |
