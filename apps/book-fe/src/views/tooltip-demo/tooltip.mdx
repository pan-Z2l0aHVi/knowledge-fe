### 基本用法

export function BasicUsage() {
  return (
    <Space size="large">
      <Tooltip title="Icon tooltip">
        <Button circle>
          <TbSearch />
        </Button>
      </Tooltip>
      <Tooltip title="Button tooltip">
        <Button primary>Primary button</Button>
      </Tooltip>
      <Tooltip title="Switch tooltip">
        <Switch />
      </Tooltip>
      <Tooltip title="Tooltip">
        <span>Text or anything</span>
      </Tooltip>
    </Space>
  )
}
