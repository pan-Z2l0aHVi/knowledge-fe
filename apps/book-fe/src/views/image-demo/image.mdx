      <Image canPreview toolbarVisible={false} src={picSrc} width="300px" />
    </>
  )
}

### 命令式图片查看器

export function ImageCommand() {
  return (
    <Space>
      <Button
        onClick={() => {
          Image.preview({
            url: picSrc
          })
        }}
      >
        Preview image
      </Button>
      <Button
        onClick={() => {
          Image.preview({
            url: picSrc,
            toolbarVisible: false
          })
        }}
      >
        Preview image without toolbar
      </Button>
    </Space>
  )
}

<ImageCommand />

### 图片裁剪

export function ImageCrop() {
  return (
    <input
      type="file"
      accept="image/*"
      onChange={e => {
        const [file] = e.target.files ?? []
        Image.crop({
          file,
          onCrop(result) {
            console.log('clip result: ', result)
          }
        })
      }}
    />
  )
}

<ImageCrop />

## API

### Image Props

| 属性       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| bordered    | 有边框           | `boolean`         | `false`  |
| shadow    | 有阴影           | `boolean`         | `false`  |
| header    | 头部           | `React.ReactNode`         | -  |
| footer    | 底部           | `React.ReactNode`         | -  |
| cover    | 封面           | `React.ReactNode`         | -  |
