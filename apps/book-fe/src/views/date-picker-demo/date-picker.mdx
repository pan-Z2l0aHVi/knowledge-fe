# DatePicker

## 代码演示

import { DatePicker, Divider, Space, Switch } from '@youknown/react-ui/src'
import dayjs, { Dayjs } from 'dayjs'
import { useState } from 'react'

---

### 基本用法

export function BasicUsage() {
  return (
    <Space>
      <DatePicker />
      <DatePicker defaultValue={dayjs()} />
      <DatePicker placeholder="请选择" />
      <DatePicker allowClear />
    </Space>
  )
}

## API

| 属性         | 说明             | 类型                             | 默认值                           |
| ------------ | ---------------- | -------------------------------- | -------------------------------- |
| size         | 大小             | `'small' \| 'medium' \| 'large'` | `'medium'`                       |
| disabled     | 禁用态           | `boolean`                        | `false`                          |
| allowClear   | 是否展示清空按钮 | `boolean`                        | `false`                          |
| icon         | 图标             | `React.ReactNode`                | `<TbCalendar />`                 |
| placeholder  | 占位符           | `string`                         | `t('react_ui.placeholder.date')` |
| defaultValue | 默认值           | `Dayjs`                          | `dayjs(null)`                    |
| value        | 封面             | `Dayjs`                          | -                                |
| onChange     | 值变化回调函数   | `(value: Dayjs) => void`         | -                                |
