# Card

## 代码演示

import { Button, Divider, Progress, Space } from '@youknown/react-ui/src'
import { useState } from 'react'
import { TbMinus, TbPlus } from 'react-icons/tb'

---

### 基本用法

export function BasicUsage() {
  const [percent, setPercent] = useState(70)
  return (
    <>
      <Space>
        <Button
          circle
          disabled={percent <= 0}
          onClick={() => {
            setPercent(p => p - 5)
          }}
        >
          <TbMinus />
        </Button>
        <Button
          circle
          disabled={percent >= 100}
          onClick={() => {
            setPercent(p => p + 5)
          }}
        >
          <TbPlus />
        </Button>
      </Space>
      <Space direction="vertical" className="w-400px">
        <Progress size="small" defaultPercent={70} percent={percent} />
        <Progress size="medium" defaultPercent={70} percent={percent} />
        <Progress size="large" defaultPercent={70} percent={percent} />
        <Progress suffix="/100" defaultPercent={70} percent={percent} />
        <Progress format={null} defaultPercent={70} percent={percent} suffix="/100" />
        <Progress
          format={({ countUpRef }) => (
            <span className="color-purple">
              <span ref={countUpRef}></span>/100
            </span>
          )}
          defaultPercent={70}
          percent={percent}
          suffix={null}
        />
      </Space>
    </>
  )
}

## API

### Progress Props

| 属性       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| size    | 大小           | `'small' \| 'medium' \| 'large'`         | `'medium'`  |
| defaultPercent    | 默认百分比           | `number`         | `0`  |
| percent    | 百分比           | `number`         | `0`  |
| direction    | 方向           | `'horizontal' \| 'vertical'`         | `'horizontal'`  |
| trackColor    | 进度轨道颜色           | `string`         | -  |
| suffix    | 后缀           | `React.ReactNode`         | `'%'`  |
| format    | 格式化函数           | `(ref: RenderCounterProps) => React.ReactNode`         | -  |

### Progress Circle Props

| 属性       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| size    | 大小           | `'small' \| 'medium' \| 'large'`         | `'medium'`  |
| defaultMolecule    | 默认百分比           | `number`         | `0`  |
| molecule    | 分子           | `number`         | `0`  |
| denominator    | 分母           | `number`         | `100`  |
| suffix    | 后缀           | `React.ReactNode`         | `'%'`  |
| format    | 格式化函数           | `((props: RenderCounterProps) => ReactNode) \| null`         | -  |
