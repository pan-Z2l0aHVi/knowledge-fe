### 头像大小

export function AvatarSize() {
  const [size, setSize] = useState('large')
  return (
    <>
      <Tabs
        type="segment"
        size="small"
        tabList={[
          { key: 'small', name: 'Small' },
          { key: 'medium', name: 'Medium' },
          { key: 'large', name: 'Large' }
        ]}
        value={size}
        onChange={setSize}
      />
      <Divider />
      <Space>
        <Avatar size={size} src={picSrc} />
        <Avatar size={size}>
          <TbUser size={20} />
        </Avatar>
        <Avatar size={size}>
          <span className='text-12px'>Text</span>
        </Avatar>
      </Space>
      <Divider />
      <Space>
        <Avatar.Group size={size}>
          <Avatar round src={picSrc} />
          <Avatar round src={picSrc} />
          <Avatar round src={picSrc} />
          <Avatar round src={picSrc} />
          <Avatar round src={picSrc} />
        </Avatar.Group>
      </Space>
    </>
  )
}

<AvatarSize />

## API

| 属性       | 说明               | 类型              | 默认值   |
| ---------- | ------------------ | ----------------- | -------- |
| bordered    | 有边框           | `boolean`         | `false`  |
| canPreview   | 开启图片查看器           | `boolean`         | `false`  |
| size       | 大小               | `number \| 'small' \| 'medium' \| 'large'` | `'medium'` |
| round      | 大圆角形状         | `boolean`         | `false`  |
| src    | 图像地址         | `string`         | `''`  |
| color    | 背景色         | `string`         | `'var(--ui-bg-3)'`  |
| badge    | 角标         | `React.ReactNode`         | -  |
